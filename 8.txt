#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int num1, num2;
    cout<<"Introduce primer numero: "<<endl;
    cin>>num1;

    cout<<"Introduce segundo numero: "<<endl;
    cin>>num2;


    cout<<"suma: "<<num1+num2<<endl;
    cout<<"diferencia: "<<num1-num2<<endl;
    cout<<"producto: "<<num1%num2<<endl;
    cout<<"cociente: "<<num1/num2<<endl;

    return 0;
}
